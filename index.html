const SHEET_ID = '1wLANvdG6ga5xAEoWe7Qd8rWO1snfU-L0fnslMsKIPxw'; const CSV_URL = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/export?format=csv&gid=0`; let ALL_DATA = []; let currentMonth = ""; const MONTH_NAMES = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"]; // Robust CSV Parser function parseCSV(text) { const lines = text.split(/\r?\n/); return lines.map(line => { const result = []; let cur = '', inQuote = false; for (let char of line) { if (char === '"') inQuote = !inQuote; else if (char === ',' && !inQuote) { result.push(cur); cur = ''; } else cur += char; } result.push(cur); return result; }); } async function init() { try { const response = await fetch(CSV_URL); const text = await response.text(); const rows = parseCSV(text); ALL_DATA = rows.slice(1).map(row => { const dateRaw = row[3] || ""; const d = new Date(dateRaw); // Generate Month Key (YYYY-MM) const monthKey = !isNaN(d.getTime()) ? `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}` : null; // Calculate Response Time if column J is a Date let respHours = null; if (row[9] && row[9].includes('2026')) { const replyDate = new Date(row[9]); respHours = Math.round(Math.abs(replyDate - d) / 36e5); // ms to hours } return { link: row[0], date: !isNaN(d.getTime()) ? d.toLocaleDateString('en-GB') : dateRaw, monthKey: monthKey, stars: parseInt(row[4]) || 0, content: row[6] || "", replyStatus: row[8] ? row[8].toLowerCase().trim() : "", responseTime: respHours, lang: row[10] || "Unknown" }; }).filter(d => d.monthKey && d.monthKey !== "NaN-NaN"); // Sort unique months const monthKeys = [...new Set(ALL_DATA.map(d => d.monthKey))].sort().reverse(); const select = document.getElementById('month-select'); select.innerHTML = ""; // Clear monthKeys.forEach(key => { const [y, m] = key.split('-'); const opt = document.createElement('option'); opt.value = key; opt.text = `${MONTH_NAMES[parseInt(m)-1]} ${y}`; select.appendChild(opt); }); currentMonth = monthKeys[0]; updateDashboard(); } catch (e) { console.error("Dashboard Load Error:", e); } } function updateDashboard() { currentMonth = document.getElementById('month-select').value; const filtered = ALL_DATA.filter(d => d.monthKey === currentMonth); // Calculate Prev Month for Comparison let [y, m] = currentMonth.split('-').map(Number); let prevKey = m === 1 ? `${y-1}-12` : `${y}-${String(m-1).padStart(2,'0')}`; const prevData = ALL_DATA.filter(d => d.monthKey === prevKey); renderKPIs(filtered, prevData); renderLangChart(filtered); } function renderKPIs(curr, prev) { const avg = curr.length ? (curr.reduce((s, r) => s + r.stars, 0) / curr.length).toFixed(1) : "0.0"; const prevAvg = prev.length ? (prev.reduce((s, r) => s + r.stars, 0) / prev.length).toFixed(1) : "0.0"; const diff = (avg - prevAvg).toFixed(1); document.getElementById('kpi-container').innerHTML = `
TrustScore
${avg} ★
${diff >= 0 ? '▲' : '▼'} ${Math.abs(diff)}
Positive
${curr.filter(r=>r.stars>=4).length}
Negative
${curr.filter(r=>r.stars<=2).length}
Unreplied
${curr.filter(r=>r.replyStatus!=='fixed').length}
Total Volume
${curr.length}
`; } function openDrill(title, filterType) { const filtered = ALL_DATA.filter(d => d.monthKey === currentMonth); let list = []; if(filterType === 'all') list = filtered; else if(filterType === 'pos') list = filtered.filter(r => r.stars >= 4); else if(filterType === 'neg') list = filtered.filter(r => r.stars <= 2); else if(filterType === 'unreplied') list = filtered.filter(r => r.replyStatus !== 'fixed'); else list = filtered.filter(r => r.lang === filterType); document.getElementById('modal-title').textContent = `${title} (${list.length})`; const body = document.getElementById('modal-body'); body.innerHTML = list.map(r => `
${'★'.repeat(r.stars)}${'☆'.repeat(5-r.stars)}
"${r.content.substring(0, 250)}..."
📅 ${r.date} | Market: ${r.lang.toUpperCase()} View Review ↗
`).join('') || "No reviews found."; document.getElementById('overlay').classList.add('open'); } function closeModal() { document.getElementById('overlay').classList.remove('open'); } function renderLangChart(data) { const langs = {}; data.forEach(r => { langs[r.lang] = (langs[r.lang] || 0) + 1; }); const ctx = document.getElementById('langChart').getContext('2d'); if (window.myChart) window.myChart.destroy(); window.myChart = new Chart(ctx, { type: 'doughnut', data: { labels: Object.keys(langs).map(l => l.toUpperCase()), datasets: [{ data: Object.values(langs), backgroundColor: ['#00b67a', '#191919', '#8A9BBC', '#E3E9F4', '#3B82F6'] }] }, options: { responsive: true, maintainAspectRatio: false, onClick: (evt, item) => { if (item.length > 0) { const index = item[0].index; const label = window.myChart.data.labels[index].toLowerCase(); openDrill(`Market: ${label.toUpperCase()}`, label); } } } }); } init();
